int exGcd(int a,int b,int &x,int &y){
	if(b==0){
	x=1;
	y=0;
	return a;//设立递归边界
	}
	int g=exGcd(b,a%b,x,y);//递归计算
	int temp =x;//存放x
	x=y;//x=y(old)
	y=temp-a/b*y;//更新y=x(old)-a/b*y(old);
	return g;//g就是gcd
}
