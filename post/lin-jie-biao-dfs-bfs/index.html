邻接表 DFS BFS

2020-12-20 2 min read # 算法竞赛笔记

邻接表状态下对图的DFS与BFS操作

#include<bits/stdc++.h>
using namespace std;
bool visite[100001];
int n,m;
vector<int>G[100001];
void dfs(int x,int r){
	visite[x]=true;
	cout<<x<<" ";
	if(r==n)return ;
	for(int i=0;i<G[x].size();i++){
		if(!visite[G[x][i]]){
			dfs(G[x][i],++r);
		}
	}
}
queue<int>v;
void bfs(int x){
	memset(visite,0,sizeof(visite));
	visite[x]=1;
	v.push(x);
	while(!v.empty()){
		int xx=v.front();
		v.pop();
		cout<<xx<<" ";
		for(int i=0;i<G[xx].size();i++){
			if(!visite[G[xx][i]]){
				visite[G[xx][i]]=1;
				v.push(G[xx][i]);
			}
		}
	}
}
int main(){
	cin>>n>>m;
	for(int i=1;i<=m;i++){
		int u,v;
		scanf("%d %d",&u,&v);
		G[u].push_back(v);
	}
	for(int i=1;i<=n;i++){
		sort(G[i].begin(),G[i].end());
		
	}
	dfs(1,0);
	cout<<endl;
	bfs(1);
	cout<<endl;
	return 0;
	
}
